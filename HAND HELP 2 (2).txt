            const overlayContainer = document.getElementById('immersiveAuthGateway');
            overlayContainer.style.display = 'flex';
            
            // Micro-task timeout to ensure CSS transitions trigger correctly
            setTimeout(() => {
                overlayContainer.classList.add('active');
            }, 10);
            
            // Freeze background viewport completely
            document.body.style.overflow = 'hidden';
            document.documentElement.style.overflow = 'hidden';
        }

        // Window Deactivation & Viewport Normalization Flow
        function terminateAuthBridge() {
            const overlayContainer = document.getElementById('immersiveAuthGateway');
            overlayContainer.classList.remove('active');
            
            setTimeout(() => {
                overlayContainer.style.display = 'none';
                // Restore body and document scroll components
                document.body.style.overflow = 'auto';
                document.documentElement.style.overflow = 'auto';
            }, 400);
        }
    </script>
